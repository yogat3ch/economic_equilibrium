  if (nrow(.x)) {
    # Use an empirical cumulative distribution function
    p_fn <- ecdf(.x$net_worth)
    list(data = dplyr::mutate(.x, p = p_fn(net_worth)),
         ecdf = p_fn)

    .x
})
